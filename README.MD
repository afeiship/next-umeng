# next-umeng
> Umeng statistic helper based on next.

## usage:

```js
var NxUmeng = require('next-umeng);
var $umeng = new NxUmeng();
$umeng.track(event_name, properties, callback, timeout);
$umeng.trackLinks(query , event_name , properties, timeout);
```
